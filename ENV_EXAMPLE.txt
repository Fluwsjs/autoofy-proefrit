# Environment Variables Voorbeeld
# Kopieer dit naar .env.local en vul de waarden in

# NextAuth
# DEVELOPMENT: http://localhost:3000
# PRODUCTION: https://proefrit-autoofy.nl (je eigen domein!)
NEXTAUTH_URL=http://localhost:3000
NEXTAUTH_SECRET=your-secret-key-here

# ⚠️ BELANGRIJK VOOR EMAIL LINKS:
# - Development: NEXTAUTH_URL=http://localhost:3000 (email links wijzen naar localhost)
# - Productie: NEXTAUTH_URL=https://proefrit-autoofy.nl (email links wijzen naar je domein)
# Email verificatie en wachtwoord reset links gebruiken deze URL!

# Email Service (Kies Resend OF SMTP)

SMTP_PASS=your-email-password

# FROM settings voor SMTP
